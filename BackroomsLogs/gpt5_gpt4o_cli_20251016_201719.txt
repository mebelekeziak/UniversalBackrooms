def info(p):
    st = p.stat()
    h = hashlib.sha256()
    if p.is_file():

        sha = h.hexdigest()
    else:

    return dict(path=str(p.relative_to(base)), size=st.st_size, mtime=int(st.st_mtime), sha256=sha, type='dir' if p.is_dir() else 'file')
